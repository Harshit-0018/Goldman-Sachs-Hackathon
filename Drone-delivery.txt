// Input : line 1 = T; then per case: line A = root name, line B = compact JSON.
// Output: per-case blocks joined by a line `---`, ending with one '\n'.

#include <bits/stdc++.h>
using namespace std;

// ===== Minimal JSON Parser =====
struct JsonVal {
    enum Type { Null, Bool, Num, Str, Arr, Obj } type;
    bool b; double n; string s;
    vector<JsonVal> arr;
    vector<pair<string,JsonVal>> obj;
};

static size_t pos_g;
static const string* src_g;

void skipWS() { while (pos_g < src_g->size() && isspace((*src_g)[pos_g])) pos_g++; }

JsonVal parseVal();

JsonVal parseStr() {
    JsonVal v; v.type = JsonVal::Str;
    pos_g++; // skip opening "
    while ((*src_g)[pos_g] != '"') {
        if ((*src_g)[pos_g] == '\\') {
            pos_g++;
            char c = (*src_g)[pos_g++];
            if (c=='n') v.s+='\n';
            else if(c=='t') v.s+='\t';
            else if(c=='r') v.s+='\r';
            else v.s+=c;
        } else v.s += (*src_g)[pos_g++];
    }
    pos_g++; return v;
}

JsonVal parseNum() {
    JsonVal v; v.type = JsonVal::Num;
    size_t st = pos_g;
    if ((*src_g)[pos_g]=='-') pos_g++;
    while (pos_g < src_g->size() && (isdigit((*src_g)[pos_g])||(*src_g)[pos_g]=='.'||
           (*src_g)[pos_g]=='e'||(*src_g)[pos_g]=='E'||(*src_g)[pos_g]=='+'||(*src_g)[pos_g]=='-'))
        pos_g++;
    v.n = stod(src_g->substr(st, pos_g-st)); return v;
}

JsonVal parseArr() {
    JsonVal v; v.type = JsonVal::Arr; pos_g++; skipWS();
    if ((*src_g)[pos_g]==']') { pos_g++; return v; }
    while (true) {
        v.arr.push_back(parseVal()); skipWS();
        if ((*src_g)[pos_g]==']') { pos_g++; break; }
        pos_g++; skipWS();
    }
    return v;
}

JsonVal parseObj() {
    JsonVal v; v.type = JsonVal::Obj; pos_g++; skipWS();
    if ((*src_g)[pos_g]=='}') { pos_g++; return v; }
    while (true) {
        JsonVal ks = parseStr(); skipWS(); pos_g++; skipWS(); // skip ':'
        JsonVal val = parseVal();
        v.obj.push_back({ks.s, val}); skipWS();
        if ((*src_g)[pos_g]=='}') { pos_g++; break; }
        pos_g++; skipWS();
    }
    return v;
}

JsonVal parseVal() {
    skipWS(); char c = (*src_g)[pos_g];
    if (c=='"') return parseStr();
    if (c=='{') return parseObj();
    if (c=='[') return parseArr();
    if (c=='-'||isdigit(c)) return parseNum();
    if (src_g->substr(pos_g,4)=="null")  { pos_g+=4; JsonVal v; v.type=JsonVal::Null; return v; }
    if (src_g->substr(pos_g,4)=="true")  { pos_g+=4; JsonVal v; v.type=JsonVal::Bool; v.b=true; return v; }
    if (src_g->substr(pos_g,5)=="false") { pos_g+=5; JsonVal v; v.type=JsonVal::Bool; v.b=false; return v; }
    throw runtime_error("bad json");
}

JsonVal parseJSON(const string& s) {
    src_g = &s; pos_g = 0; return parseVal();
}

// ===== Type System =====
struct Interface {
    string name;
    map<string, pair<bool,string>> props; // key -> (optional, typestring)
};

static map<string,Interface> interfaces;
static set<string> usedNames;
static map<string,string> pathToName;

string capitalize(const string& k) {
    return string(1, toupper(k[0])) + k.substr(1);
}

string assignName(const string& path, const string& baseName) {
    if (pathToName.count(path)) return pathToName[path];
    string candidate = baseName;
    if (!usedNames.count(candidate)) {
        usedNames.insert(candidate);
        pathToName[path] = candidate;
        return candidate;
    }
    int suffix = 2;
    while (usedNames.count(candidate + to_string(suffix))) suffix++;
    string chosen = candidate + to_string(suffix);
    usedNames.insert(chosen);
    pathToName[path] = chosen;
    return chosen;
}

// Forward declarations
string computeArrayTypeStr(const string& path, const string& keyForNaming, vector<JsonVal*>& allElems);
string computeTypeStr(const string& path, const string& key, vector<JsonVal*>& vals);

string mergeObjects(const string& path, const string& baseName, vector<JsonVal*>& objVals) {
    map<string, vector<JsonVal*>> keyVals;
    int totalObjs = (int)objVals.size();
    map<string, int> keyCount;

    for (auto* ov : objVals) {
        set<string> seen;
        for (auto& [k,v] : ov->obj) {
            keyVals[k].push_back(const_cast<JsonVal*>(&v));
            if (!seen.count(k)) { keyCount[k]++; seen.insert(k); }
        }
    }

    string ifName;
    if (pathToName.count(path)) {
        ifName = pathToName[path];
    } else {
        ifName = assignName(path, baseName);
    }

    Interface& iface = interfaces[ifName];
    iface.name = ifName;

    // Process keys in sorted ASCII order (depth-first traversal for name assignment)
    vector<string> sortedKeys;
    for (auto& [k,_] : keyVals) sortedKeys.push_back(k);
    sort(sortedKeys.begin(), sortedKeys.end());

    for (auto& k : sortedKeys) {
        bool optional = (keyCount[k] < totalObjs);
        auto& vals = keyVals[k];
        string childPath = path + "." + k;
        string typeStr = computeTypeStr(childPath, k, vals);
        iface.props[k] = {optional, typeStr};
    }

    return ifName;
}

string computeTypeStr(const string& path, const string& key, vector<JsonVal*>& vals) {
    vector<JsonVal*> arrVals, objVals, primVals;
    for (auto* v : vals) {
        if (v->type == JsonVal::Arr) arrVals.push_back(v);
        else if (v->type == JsonVal::Obj) objVals.push_back(v);
        else primVals.push_back(v);
    }

    set<string> typeStrings;

    if (!arrVals.empty()) {
        vector<JsonVal*> allElems;
        for (auto* av : arrVals)
            for (auto& e : av->arr) allElems.push_back(const_cast<JsonVal*>(&e));
        typeStrings.insert(computeArrayTypeStr(path, key, allElems));
    }

    if (!objVals.empty()) {
        string baseName = capitalize(key);
        string ifName = mergeObjects(path, baseName, objVals);
        typeStrings.insert(ifName);
    }

    for (auto* v : primVals) {
        if (v->type == JsonVal::Null)  typeStrings.insert("null");
        else if (v->type == JsonVal::Bool) typeStrings.insert("boolean");
        else if (v->type == JsonVal::Num)  typeStrings.insert("number");
        else if (v->type == JsonVal::Str)  typeStrings.insert("string");
    }

    if (typeStrings.size() == 1) return *typeStrings.begin();
    vector<string> ts(typeStrings.begin(), typeStrings.end());
    sort(ts.begin(), ts.end());
    string result;
    for (int i=0;i<(int)ts.size();i++) { if(i) result+=" | "; result+=ts[i]; }
    return result;
}

string computeArrayTypeStr(const string& path, const string& keyForNaming, vector<JsonVal*>& allElems) {
    if (allElems.empty()) return "unknown[]";

    vector<JsonVal*> objElems, primElems;
    for (auto* e : allElems) {
        if (e->type == JsonVal::Obj) objElems.push_back(e);
        else primElems.push_back(e);
    }

    set<string> elemTypes;

    if (!objElems.empty()) {
        string baseName = capitalize(keyForNaming);
        string ifName = mergeObjects(path + "[]", baseName, objElems);
        elemTypes.insert(ifName);
    }

    for (auto* e : primElems) {
        if (e->type == JsonVal::Null)  elemTypes.insert("null");
        else if (e->type == JsonVal::Bool) elemTypes.insert("boolean");
        else if (e->type == JsonVal::Num)  elemTypes.insert("number");
        else if (e->type == JsonVal::Str)  elemTypes.insert("string");
    }

    if (elemTypes.size() == 1) return *elemTypes.begin() + "[]";
    vector<string> et(elemTypes.begin(), elemTypes.end());
    sort(et.begin(), et.end());
    string union_str;
    for (int i=0;i<(int)et.size();i++) { if(i) union_str+=" | "; union_str+=et[i]; }
    return "(" + union_str + ")[]";
}

static string solve(const string& rootName, const string& jsonText) {
    // Reset global state per test case
    interfaces.clear();
    usedNames.clear();
    pathToName.clear();

    // Root name is reserved first (always gets unsuffixed name)
    usedNames.insert(rootName);
    pathToName["$"] = rootName;

    JsonVal root = parseJSON(jsonText);

    if (root.arr.empty()) {
        Interface iface; iface.name = rootName;
        interfaces[rootName] = iface;
    } else {
        vector<JsonVal*> topObjs;
        for (auto& e : root.arr) topObjs.push_back(const_cast<JsonVal*>(&e));
        mergeObjects("$", rootName, topObjs);
    }

    // Sort interfaces by name (case-sensitive ASCII order)
    vector<string> ifNames;
    for (auto& [k,_] : interfaces) ifNames.push_back(k);
    sort(ifNames.begin(), ifNames.end());

    string out;
    for (int i=0;i<(int)ifNames.size();i++) {
        if (i) out += "\n\n";
        auto& iface = interfaces[ifNames[i]];
        if (iface.props.empty()) {
            out += "export interface " + iface.name + " {}";
        } else {
            out += "export interface " + iface.name + " {\n";
            for (auto& [k, ov] : iface.props) {
                auto& [opt, ts] = ov;
                out += "  " + k + (opt?"?":"") + ": " + ts + ";\n";
            }
            out += "}";
        }
    }
    return out;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;
    cin.ignore();

    string out;
    for (int i = 0; i < t; i++) {
        string rootName, jsonText;
        getline(cin, rootName);
        getline(cin, jsonText);
        if (i > 0) out += "\n---\n";
        out += solve(rootName, jsonText);
    }
    out += '\n';

    cout << out;
    return 0;
}
